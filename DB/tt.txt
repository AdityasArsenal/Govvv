CREATE TABLE public.monthly_sheet_data (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  sheet_name text NOT NULL,
  month text NOT NULL,
  data jsonb,
  last_saved timestamp with time zone DEFAULT now(),
  created_at timestamp with time zone DEFAULT now(),
  user_id uuid,
  CONSTRAINT monthly_sheet_data_pkey PRIMARY KEY (id),
  CONSTRAINT monthly_sheet_data_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
