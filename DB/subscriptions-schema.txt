create table public.subscriptions ( id uuid not null default gen_random_uuid (), user_id uuid not null, status text not null, start_date timestamp with time zone null, end_date timestamp with time zone null, plan_id text null, phonepe_transaction_id text null, created_at timestamp with time zone null default now(), updated_at timestamp with time zone null default now(), constraint subscriptions_pkey primary key (id), constraint subscriptions_user_id_fkey foreign KEY (user_id) references auth.users (id) on delete CASCADE ) TABLESPACE pg_default;
